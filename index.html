General-Purpose Types

std::future::Future
futures for async, await
std::option::Option<T>
Some(T) or None
std::result::Result<T,E>
Ok(T) or Err(E)
std::boxed::Box<T>
simplest heap allocation
Box::new(value) and let value: u8 = *boxed
std::rc::Rc<T>
single-threaded, non-atomic-reference-counted pointer
std::rc::Weak<T>
Rc for non-owning references
std::sync::Arc<T>
thread-safe, atomic-reference-counted poitner
